# 商品数据结构

商品数据文件使用 JSON 格式，根对象必须包含 `info`、`size_table`、`quick_size`。可选字段 `colors` 用于保存拆分后的颜色顺序。

## JSON 结构约束

```json
{
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "title": "Visual Detail Product Data",
  "type": "object",
  "required": ["info", "size_table", "quick_size"],
  "additionalProperties": true,
  "properties": {
    "info": {
      "type": "object",
      "required": ["名称", "货号", "材质", "颜色", "尺码"],
      "properties": {
        "名称": { "type": "string" },
        "货号": { "type": "string" },
        "材质": { "type": "string" },
        "颜色": { "type": "string" },
        "尺码": { "type": "string" }
      },
      "additionalProperties": true
    },
    "colors": {
      "type": "array",
      "items": { "type": "string" }
    },
    "size_table": {
      "type": "object",
      "required": ["headers", "rows"],
      "properties": {
        "headers": {
          "type": "array",
          "minItems": 1,
          "items": { "type": "string" }
        },
        "rows": {
          "type": "array",
          "items": {
            "type": "array",
            "items": { "type": ["string", "number", "null"] }
          }
        }
      }
    },
    "quick_size": {
      "type": "object",
      "required": ["columns", "rows", "data"],
      "properties": {
        "columns": {
          "type": "array",
          "minItems": 1,
          "items": { "type": "string" }
        },
        "rows": {
          "type": "array",
          "minItems": 1,
          "items": { "type": "string" }
        },
        "data": {
          "type": "array",
          "items": {
            "type": "array",
            "items": { "type": ["string", "number", "null"] }
          }
        }
      }
    }
  }
}
```
